<script>
  function onClickArrow(e) {
      e.preventDefault();
      let items = this.parentNode.querySelector(':scope > .items');
      let itemsAreHidden = items && items.hidden;
      items.hidden = !itemsAreHidden;
      this.classList.remove('to-bottom', 'to-right');
      this.classList.add('to-' + (itemsAreHidden ? 'bottom' : 'right'));
      this.parentNode.querySelector('.icon > path').setAttribute('d', itemsAreHidden ? 'M19 20H4a2 2 0 0 1-2-2V6c0-1.11.89-2 2-2h6l2 2h7a2 2 0 0 1 2 2H4v10l2.14-8h17.07l-2.28 8.5c-.23.87-1.01 1.5-1.93 1.5' : 'M10 4H4c-1.11 0-2 .89-2 2v12a2 2 0 0 0 2 2h16a2 2 0 0 0 2-2V8a2 2 0 0 0-2-2h-8z');
  }
  document.querySelectorAll('.item > .arrow').forEach(arrow => {
      arrow.addEventListener('click', onClickArrow, false);
  });
</script>
